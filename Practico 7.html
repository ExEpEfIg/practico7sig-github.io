In[]:
# Esta funcin genera una variable que contenga los archivos tif que estan en ese directorio.
lista=glob.glob(r"D:\Documentos\Magister en Recursos Hdricos\SIG_TIC\Semana_14\landsat/ndwi/*.tif")

# Crear el directorio de salida 
carpetaSalida = os.path.join(r"D:\Documentos\Magister en Recursos Hdricos\SIG_TIC\Semana_14\landsat/ndwi", "ndwi_clases")
os.makedirs(carpetaSalida, exist_ok=True)

#Se crea variable que abre la lista anterior contenida con archivos .tif   
metaRaster=rasterio.open(lista[0])


# Iterar sobre cada archivo TIFF en la lista
for image in lista:
    # Abrir el archivo raster y leer la primera banda (NDWI en este caso)
    with rasterio.open(image) as metaRaster:
        arrayRaster = metaRaster.read(1)  # Leer la primera banda como array de Numpy

        # Crear una copia del array para clasificar el NDWI
        NDWI_clases = np.zeros_like(arrayRaster, dtype=np.int32)  # Inicializar array de clases

        # Clasificar los valores del NDWI
        NDWI_clases[np.where(arrayRaster < 0)] = 0  # Valores menores a 0 clasificados como 0
        NDWI_clases[np.where(arrayRaster >= 0)] = 1  # Valores mayores o iguales a 0 clasificados como 1

        # Definir el perfil (metadata) para la nueva imagen
        profile = metaRaster.profile
        profile["dtype"] = "int32"  # Cambiar el tipo de datos a int32
        profile["count"] = 1  # Nmero de bandas

        # Crear la ruta de salida para el archivo clasificado
        output_path = os.path.join(carpetaSalida, os.path.basename(image)[:-4] + "_clases.tif")

        # Exportar la imagen clasificada en formato TIFF
        with rasterio.open(output_path, 'w', **profile) as export:
            export.write(NDWI_clases.astype(rasterio.int32), 1)  # Escribir la primera banda

print("Imgenes exportadas con xito.")

4. Grfico que muestre tendencia temporal de la superficie cubierta por agua

4.1 Preparacin de los datos; obtencin de dataframe con los datos a plotear

In[]:
import glob
import rasterio
import pandas as pd
import matplotlib.pyplot as plt
import matplotlib.ticker as ticker

# Directorio donde se encuentran los archivos TIF clasificados por aos
folder_path = r"D:\Documentos\Magister en Recursos Hdricos\SIG_TIC\Semana_14\landsat\ndwi\ndwi_clases"

# Obtener la lista de archivos TIF en el directorio
lista_archivos = glob.glob(folder_path + r"\*.tif")

# Inicializar una lista para almacenar los datos
data = []

# Iterar sobre cada archivo TIF para procesar la informacin
for archivo in lista_archivos:
    # Obtener el ao del nombre del archivo
    ao = int(archivo.split("\\")[-1].split("_")[0])

    with rasterio.open(archivo) as src:
        # Leer la primera banda como array de numpy
        array_ndwi = src.read(1)

        # Calcular el tamao de pxel en metros cuadrados
        pixel_size = src.transform[0]
        
        # Contar pxeles clasificados como agua
        area_agua_pixels = (array_ndwi == 1).sum()

        # Calcular el rea en kilmetros cuadrados
        area_agua_km2 = (pixel_size ** 2) * area_agua_pixels / 1e6

        # Agregar los datos a la lista
        data.append({"Ao": ao, "rea_km2": area_agua_km2})

# Crear un DataFrame a partir de la lista de datos
df_series_tiempo = pd.DataFrame(data)
